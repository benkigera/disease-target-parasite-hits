const COMPOSITE = (d) => {
  return (0.55 * d.eff + 0.25 * d.pre + 0.10 * (7 - d.pip) + 0.05 * d.saf + 0.05 * d.con);
};

const AREAS = [...new Set(INDICATIONS.map(d => d.area))].sort();

const COLORS = {
  parasite: "#1D9E75",
  noParasite: "#888780",
  selected: "#D85A30",
};

const ScoreDot = ({ value, max = 7, color }) => {
  const pct = value / max;
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 6 }}>
      <div style={{ flex: 1, height: 5, background: "var(--color-background-secondary)", borderRadius: 3, overflow: "hidden" }}>
        <div style={{ width: `${pct * 100}%`, height: "100%", background: color || "var(--color-text-secondary)", borderRadius: 3 }} />
      </div>
      <span style={{ fontSize: 12, fontWeight: 500, minWidth: 14, color: "var(--color-text-primary)" }}>{value}</span>
    </div>
  );
};

const ScoreRow = ({ label, value, color }) => (
  <div style={{ marginBottom: 6 }}>
    <div style={{ fontSize: 11, color: "var(--color-text-secondary)", marginBottom: 3, textTransform: "uppercase", letterSpacing: "0.06em" }}>{label}</div>
    <ScoreDot value={value} color={color} />
  </div>
);

const CustomTooltip = ({ active, payload }) => {
  if (!active || !payload?.length) return null;
  const d = payload[0]?.payload;
  if (!d) return null;
  return (
    <div style={{
      background: "var(--color-background-primary)",
      border: "0.5px solid var(--color-border-secondary)",
      borderRadius: "var(--border-radius-md)",
      padding: "10px 14px",
      fontSize: 12,
      maxWidth: 220,
      pointerEvents: "none",
    }}>
      <div style={{ fontWeight: 500, marginBottom: 4, color: "var(--color-text-primary)" }}>{d.condition}</div>
      <div style={{ color: "var(--color-text-secondary)", marginBottom: 6 }}>{d.area}</div>
      <div style={{ display: "flex", gap: 16 }}>
        <div><span style={{ color: "var(--color-text-secondary)" }}>EFF </span><span style={{ fontWeight: 500 }}>{d.eff}/7</span></div>
        <div><span style={{ color: "var(--color-text-secondary)" }}>PIP </span><span style={{ fontWeight: 500 }}>{d.pip}/7</span></div>
      </div>
      {d.parasite && (
        <div style={{ marginTop: 6, padding: "4px 8px", background: "#E1F5EE", borderRadius: 4, color: "#0F6E56", fontSize: 11, fontWeight: 500 }}>
          Parasite molecule candidate
        </div>
      )}
    </div>
  );
};

                  </div>

            </div>
          ) : (
            <div style={{ display: "flex", flexDirection: "column", justifyContent: "center", alignItems: "center", height: "100%", minHeight: 300, color: "var(--color-text-secondary)", textAlign: "center", gap: 8 }}>
              <i className="ti ti-cursor-text" style={{ fontSize: 28 }} aria-hidden="true" />
              <div style={{ fontSize: 13 }}>Click a bubble or row<br />to inspect the indication</div>
            </div>
          )}
        </div>
      </div>

      <div style={{ border: "0.5px solid var(--color-border-tertiary)", borderRadius: "var(--border-radius-lg)", overflow: "hidden" }}>
        <table style={{ width: "100%", borderCollapse: "collapse", fontSize: 13, tableLayout: "fixed" }}>
          <thead>
            <tr style={{ background: "var(--color-background-secondary)" }}>
              <th style={{ padding: "10px 14px", textAlign: "left", fontWeight: 500, fontSize: 11, color: "var(--color-text-secondary)", textTransform: "uppercase", letterSpacing: "0.06em", width: "34%" }}>Indication</th>
              <th style={{ padding: "10px 8px", textAlign: "center", fontWeight: 500, fontSize: 11, color: "var(--color-text-secondary)", textTransform: "uppercase", letterSpacing: "0.06em", width: "10%" }}>EFF</th>
              <th style={{ padding: "10px 8px", textAlign: "center", fontWeight: 500, fontSize: 11, color: "var(--color-text-secondary)", textTransform: "uppercase", letterSpacing: "0.06em", width: "10%" }}>PRE</th>
              <th style={{ padding: "10px 8px", textAlign: "center", fontWeight: 500, fontSize: 11, color: "var(--color-text-secondary)", textTransform: "uppercase", letterSpacing: "0.06em", width: "10%" }}>PIP</th>
              <th style={{ padding: "10px 8px", textAlign: "center", fontWeight: 500, fontSize: 11, color: "var(--color-text-secondary)", textTransform: "uppercase", letterSpacing: "0.06em", width: "12%" }}>Score</th>
              <th style={{ padding: "10px 14px", textAlign: "left", fontWeight: 500, fontSize: 11, color: "var(--color-text-secondary)", textTransform: "uppercase", letterSpacing: "0.06em", width: "24%" }}>Parasite candidate</th>
            </tr>
          </thead>
          <tbody>
            {sorted.map((d, i) => {
              const score = COMPOSITE(d).toFixed(2);
              const isSelected = d.id === selectedId;
              return (
                <tr
                  key={d.id}
                  onClick={() => setSelectedId(d.id === selectedId ? null : d.id)}
                  style={{
                    borderTop: "0.5px solid var(--color-border-tertiary)",
                    background: isSelected ? "var(--color-background-secondary)" : "transparent",
                    cursor: "pointer",
                  }}
                  onMouseEnter={e => { if (!isSelected) e.currentTarget.style.background = "var(--color-background-secondary)"; }}
                  onMouseLeave={e => { if (!isSelected) e.currentTarget.style.background = "transparent"; }}
                >
                  <td style={{ padding: "10px 14px" }}>
                    <div style={{ fontWeight: isSelected ? 500 : 400 }}>{d.condition}</div>
                    <div style={{ fontSize: 11, color: "var(--color-text-secondary)", marginTop: 1 }}>{d.area} · {d.prevalence}</div>
                  </td>
                  <td style={{ padding: "10px 8px", textAlign: "center" }}>
                    <span style={{
                      display: "inline-block", padding: "2px 7px", borderRadius: 4,
                      background: d.eff >= 6 ? "#E1F5EE" : d.eff >= 5 ? "#FAEEDA" : "var(--color-background-secondary)",
                      color: d.eff >= 6 ? "#0F6E56" : d.eff >= 5 ? "#854F0B" : "var(--color-text-primary)",
                      fontWeight: 500, fontSize: 12
                    }}>{d.eff}</span>
                  </td>
                  <td style={{ padding: "10px 8px", textAlign: "center", color: "var(--color-text-primary)" }}>{d.pre}</td>
                  <td style={{ padding: "10px 8px", textAlign: "center" }}>
                    <span style={{
                      display: "inline-block", padding: "2px 7px", borderRadius: 4,
                      background: d.pip <= 3 ? "#FAECE7" : "var(--color-background-secondary)",
                      color: d.pip <= 3 ? "#993C1D" : "var(--color-text-primary)",
                      fontWeight: d.pip <= 3 ? 500 : 400, fontSize: 12
                    }}>{d.pip}</span>
                  </td>
                  <td style={{ padding: "10px 8px", textAlign: "center", fontWeight: 500 }}>{score}</td>
                  <td style={{ padding: "10px 14px" }}>
                    {d.parasite ? (
                      <span style={{ fontSize: 11, color: "#0F6E56", fontWeight: 500, display: "flex", alignItems: "center", gap: 4 }}>
                        <span style={{ width: 6, height: 6, borderRadius: "50%", background: COLORS.parasite, display: "inline-block", flexShrink: 0 }} />
                        {d.parasiteMolecule}
                      </span>
                    ) : (
                      <span style={{ fontSize: 11, color: "var(--color-text-secondary)" }}>—</span>
                    )}
                  </td>
                </tr>
              );
            })}
          </tbody>
        </table>
      </div>

      <div style={{ marginTop: "1rem", fontSize: 11, color: "var(--color-text-secondary)", lineHeight: 1.6 }}>
        Scores adapted from Convoke Unmet Needs Index methodology (EFF 55%, PRE 25%, PIP 10%, SAF 5%, CON 5%). Parasite molecule data sourced from primary literature (PubMed, Frontiers in Immunology, Scientific Reports). This is a research prototype demonstrating how Convoke's index can be augmented with orthogonal biomedical signals.
      </div>
    </div>
  );
}
